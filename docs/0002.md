### **这次升级了什么，以及为什么**

这次是从“能看、能提交”升级到“适合正式获客”的版本，核心升级有六个方面。

#### **1. 加入客户 Logo 墙**

这是企业服务官网里非常关键的一层信任背书。  
原因很简单，B2B 决策用户不会只看你自己怎么说，更看“谁在用你”。哪怕现在还不能公开真实客户名称，也建议先放“兼容平台 Logo”或“合作生态 Logo”。

#### **2. 加入 FAQ 模块**

FAQ 是最容易提升转化的模块之一。  
因为很多客户不是不感兴趣，而是还有几个顾虑没有被消除，比如：接入周期多久、老系统能不能接、上线后谁维护、出现异常怎么处理。FAQ 的作用就是把这些犹豫提前拆掉。

#### **3. 增加提交成功页 `thanks.html`**

这能让转化链路更完整。  
用户提交后不应该只是原地看到一句提示，而应该进入一个明确的“提交成功”状态页，增强可信感，也方便后续埋点统计。

#### **4. 增加 `404.html`**

这是正式商用站的基本配置。  
尤其是以后做 SEO、投放广告、发案例链接，404 页能减少跳失并把用户引导回主页。

#### **5. 补齐 SEO 基础文件**

包括：
- `robots.txt`
- `sitemap.xml`
- Open Graph meta
- canonical
- 多语言 alternate 链接

这些东西平时不显眼，但对搜索引擎抓取、页面分享展示、长期 SEO 表现很重要。

#### **6. 让表单提交流程更像“真实业务链路”**

这次我会把表单提交成功后跳转到 `thanks.html`，并把后端函数写得更适合后续接入：
- 企业微信 webhook
- 飞书 webhook
- D1 数据库存储
- 邮件通知

## **三、升级前端提交脚本，让成功后跳转感谢页**

把 `assets/app.js` 替换成下面版本。

```javascript
const form = document.getElementById("lead-form");
const statusEl = document.getElementById("form-status");

if (form) {
  form.addEventListener("submit", async (e) => {
    e.preventDefault();

    const htmlLang = document.documentElement.lang || "zh-CN";
    const isEnglish = htmlLang.toLowerCase().startsWith("en");

    const submitButton = form.querySelector('button[type="submit"]');
    const formData = new FormData(form);

    const payload = {
      company: (formData.get("company") || "").toString().trim(),
      contactName: (formData.get("contactName") || "").toString().trim(),
      phone: (formData.get("phone") || "").toString().trim(),
      requirement: (formData.get("requirement") || "").toString().trim(),
      source: window.location.pathname,
      submittedAt: new Date().toISOString()
    };

    if (!payload.phone) {
      statusEl.textContent = isEnglish
        ? "Please enter your phone number."
        : "请填写联系电话。";
      return;
    }

    submitButton.disabled = true;
    submitButton.textContent = isEnglish ? "Submitting..." : "提交中...";
    statusEl.textContent = "";

    try {
      const response = await fetch("/api/lead", {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify(payload)
      });

      const result = await response.json();

      if (!response.ok || !result.success) {
        throw new Error(result.message || (isEnglish ? "Submission failed." : "提交失败。"));
      }

      const nextUrl = isEnglish ? "/thanks.html?lang=en" : "/thanks.html";
      window.location.href = nextUrl;
    } catch (error) {
      statusEl.textContent = error.message || (isEnglish ? "Submission failed. Please try again later." : "提交失败，请稍后再试。");
    } finally {
      submitButton.disabled = false;
      submitButton.textContent = isEnglish
        ? "Submit and Get a Free Plan"
        : "提交需求，获取免费方案";
    }
  });
}
```

## **四、升级 Functions：支持 webhook 转发与更明确的数据结构**

把 `functions/api/lead.js` 替换成下面版本。

```javascript
export async function onRequestPost(context) {
  try {
    const request = context.request;
    const contentType = request.headers.get("content-type") || "";

    if (!contentType.includes("application/json")) {
      return json(
        { success: false, message: "Invalid content type. Expected application/json." },
        400
      );
    }

    const data = await request.json();

    const company = safeText(data.company, 120);
    const contactName = safeText(data.contactName, 80);
    const phone = safeText(data.phone, 50);
    const requirement = safeText(data.requirement, 2000);
    const source = safeText(data.source, 200);
    const submittedAt = safeText(data.submittedAt, 80);

    if (!phone) {
      return json({ success: false, message: "Phone is required." }, 400);
    }

    const lead = {
      company,
      contactName,
      phone,
      requirement,
      source,
      submittedAt,
      ip:
        request.headers.get("CF-Connecting-IP") ||
        request.headers.get("x-forwarded-for") ||
        "",
      userAgent: request.headers.get("user-agent") || "",
      referer: request.headers.get("referer") || ""
    };

    console.log("New lead:", JSON.stringify(lead));

    await sendToWeComIfConfigured(context, lead);
    await sendToFeishuIfConfigured(context, lead);

    return json({
      success: true,
      message: "Lead submitted successfully."
    });
  } catch (error) {
    return json(
      {
        success: false,
        message: "Server error.",
        error: error instanceof Error ? error.message : String(error)
      },
      500
    );
  }
}

async function sendToWeComIfConfigured(context, lead) {
  const webhook = context.env.WECOM_WEBHOOK_URL;
  if (!webhook) return;

  const content =
`【官网新线索】
企业名称：${lead.company || "-"}
联系人：${lead.contactName || "-"}
电话：${lead.phone || "-"}
需求：${lead.requirement || "-"}
来源：${lead.source || "-"}
时间：${lead.submittedAt || "-"}
IP：${lead.ip || "-"}
`;

  await fetch(webhook, {
    method: "POST",
    headers: {
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      msgtype: "text",
      text: { content }
    })
  });
}

async function sendToFeishuIfConfigured(context, lead) {
  const webhook = context.env.FEISHU_WEBHOOK_URL;
  if (!webhook) return;

  const content =
`官网新线索
企业名称：${lead.company || "-"}
联系人：${lead.contactName || "-"}
电话：${lead.phone || "-"}
需求：${lead.requirement || "-"}
来源：${lead.source || "-"}
时间：${lead.submittedAt || "-"}
IP：${lead.ip || "-"}
`;

  await fetch(webhook, {
    method: "POST",
    headers: {
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      msg_type: "text",
      content: { text: content }
    })
  });
}

function safeText(value, maxLength = 255) {
  if (typeof value !== "string") return "";
  return value.replace(/[\u0000-\u001F\u007F]/g, "").trim().slice(0, maxLength);
}

function json(data, status = 200) {
  return new Response(JSON.stringify(data), {
    status,
    headers: {
      "Content-Type": "application/json; charset=utf-8"
    }
  });
}
```

## **七、SEO 文件**

### **1. `robots.txt`**

你需要把域名替换成你的正式域名。

```txt
User-agent: *
Allow: /

Sitemap: https://your-domain.com/sitemap.xml
```

### **2. `sitemap.xml`**

同样，把域名替换掉。

```xml
<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
  <url>
    <loc>https://your-domain.com/</loc>
    <changefreq>weekly</changefreq>
    <priority>1.0</priority>
  </url>
  <url>
    <loc>https://your-domain.com/en.html</loc>
    <changefreq>weekly</changefreq>
    <priority>0.8</priority>
  </url>
</urlset>
```
